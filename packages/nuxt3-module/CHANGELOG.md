  change your `nuxt.config.js` from:

  ```ts
  export default defineNuxtConfig({
    ///...
    shopware: {
      shopwareEndpoint: "http://localhost:8000",
      shopwareAccessToken: "your-access-token",
    },
    ///...
  });
  ```

  to

  ```ts
  export default defineNuxtConfig({
    ///...
    runtimeConfig: {
      public: {
        shopware: {
          shopwareEndpoint: "http://localhost:8000",
          shopwareAccessToken: "your-access-token",
        },
      },
    },
    ///...
  });
  ```
